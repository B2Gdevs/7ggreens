```tsx
"use client";

import Link from "next/link";
import { ShoppingCart, Menu, X } from "lucide-react";
import { useState } from "react";
import { SiteSection } from "@portfolio/visual-context";
import { cn } from "@/lib/cn";
import { useCartCount } from "@/lib/cart-store";
import { SITE_CONSTANTS } from "@/lib/constants/site";

export function SiteHeader() {
  const [mobileOpen, setMobileOpen] = useState(false);
  const cartCount = useCartCount();

  return (
    <SiteSection cid="chrome-header" id="site-header">
      <header
        className={cn(
          "sticky top-0 z-50",
          "bg-background/95 backdrop-blur",
          "border-b border-border/40",
        )}
      >
        <div className="mx-auto flex h-16 max-w-7xl items-center justify-between px-4 sm:px-6 lg:px-8">
          {/* Logo */}
          <Link
            href="/"
            className="text-xl font-bold tracking-tight text-foreground hover:text-primary transition-colors"
            aria-label={`${SITE_CONSTANTS.brand.name} home`}
          >
            {SITE_CONSTANTS.brand.name}
          </Link>

          {/* Desktop nav */}
          <nav
            className="hidden items-center gap-6 md:flex"
            aria-label="Primary navigation"
          >
            {SITE_CONSTANTS.nav.links.map((link) => (
              <Link
                key={link.href}
                href={link.href}
                className="text-sm font-medium text-muted-foreground hover:text-foreground transition-colors"
              >
                {link.label}
              </Link>
            ))}
          </nav>

          {/* Right slot: cart + mobile toggle */}
          <div className="flex items-center gap-3">
            {/* Cart stub — Phase 04 wires Square Order state */}
            <Link
              href="/cart"
              aria-label={`Cart, ${cartCount} item${cartCount !== 1 ? "s" : ""}`}
              className="relative flex items-center gap-1.5 rounded-md px-2 py-1 text-sm font-medium text-muted-foreground hover:text-foreground transition-colors"
            >
              <ShoppingCart className="h-5 w-5" aria-hidden />
              <span className="tabular-nums">{cartCount}</span>
            </Link>

            {/* Mobile menu toggle */}
            <button
              className="flex items-center justify-center rounded-md p-2 text-muted-foreground hover:text-foreground md:hidden transition-colors"
              onClick={() => setMobileOpen((v) => !v)}
              aria-label={mobileOpen ? "Close menu" : "Open menu"}
              aria-expanded={mobileOpen}
              aria-controls="mobile-nav"
            >
              {mobileOpen ? (
                <X className="h-5 w-5" aria-hidden />
              ) : (
                <Menu className="h-5 w-5" aria-hidden />
              )}
            </button>
          </div>
        </div>

        {/* Mobile nav drawer */}
        {mobileOpen && (
          <nav
            id="mobile-nav"
            className="border-t border-border/40 bg-background px-4 pb-4 pt-2 md:hidden"
            aria-label="Mobile navigation"
          >
            <ul className="flex flex-col gap-1">
              {SITE_CONSTANTS.nav.links.map((link) => (
                <li key={link.href}>
                  <Link
                    href={link.href}
                    className="block rounded-md px-3 py-2 text-sm font-medium text-muted-foreground hover:bg-muted hover:text-foreground transition-colors"
                    onClick={() => setMobileOpen(false)}
                  >
                    {link.label}
                  </Link>
                </li>
              ))}
            </ul>
          </nav>
        )}
      </header>
    </SiteSection>
  );
}
```

```tsx
import Link from "next/link";
import { SiteSection } from "@portfolio/visual-context";
import { SITE_CONSTANTS } from "@/lib/constants/site";

export function SiteFooter() {
  const { contact, brand, copyright, nav } = SITE_CONSTANTS;

  return (
    <SiteSection cid="chrome-footer" id="site-footer">
      <footer className="border-t border-border/40 bg-card">
        <div className="mx-auto max-w-7xl px-4 py-12 sm:px-6 lg:px-8">
          <div className="grid grid-cols-1 gap-8 sm:grid-cols-2 lg:grid-cols-4">

            {/* Brand column */}
            <div className="lg:col-span-2">
              <Link
                href="/"
                className="text-lg font-bold tracking-tight text-foreground"
              >
                {brand.name}
              </Link>
              <p className="mt-2 max-w-xs text-sm text-muted-foreground">
                {brand.description}
              </p>
            </div>

            {/* Nav column */}
            <div>
              <h3 className="text-sm font-semibold text-foreground">Navigate</h3>
              <ul className="mt-3 flex flex-col gap-2">
                {nav.links.map((link) => (
                  <li key={link.href}>
                    <Link
                      href={link.href}
                      className="text-sm text-muted-foreground hover:text-foreground transition-colors"
                    >
                      {link.label}
                    </Link>
                  </li>
                ))}
              </ul>
            </div>

            {/* Contact column */}
            <div>
              <h3 className="text-sm font-semibold text-foreground">Contact</h3>
              <address className="mt-3 flex flex-col gap-2 not-italic">
                <p className="text-sm text-muted-foreground">
                  {contact.address.full}
                </p>
                {contact.phones.map((phone) => (
                  <a
                    key={phone.number}
                    href={`tel:${phone.number.replace(/-/g, "")}`}
                    className="text-sm text-muted-foreground hover:text-foreground transition-colors"
                    aria-label={`${phone.label} phone: ${phone.number}`}
                  >
                    {phone.number}
                  </a>
                ))}
              </address>

              {/* Social — placeholder until operator provides handles */}
              {(contact.social.facebook || contact.social.instagram) && (
                <div className="mt-4 flex gap-3">
                  {contact.social.facebook && (
                    <a
                      href={contact.social.facebook}
                      target="_blank"
                      rel="noopener noreferrer"
                      className="text-sm text-muted-foreground hover:text-foreground transition-colors"
                    >
                      Facebook
                    </a>
                  )}
                  {contact.social.instagram && (
                    <a
                      href={contact.social.instagram}
                      target="_blank"
                      rel="noopener noreferrer"
                      className="text-sm text-muted-foreground hover:text-foreground transition-colors"
                    >
                      Instagram
                    </a>
                  )}
                </div>
              )}
            </div>
          </div>

          {/* Bottom bar */}
          <div className="mt-10 border-t border-border/40 pt-6 text-center">
            <p className="text-xs text-muted-foreground">
              &copy; {copyright.year} {copyright.holder}. All rights reserved.
            </p>
          </div>
        </div>
      </footer>
    </SiteSection>
  );
}
```
