```css
@import "tailwindcss";

/* ─────────────────────────────────────────────
   UPAEC / 7Greens Design Tokens
   Source: RESEARCH.md Theme Tokens section
   Contrast ratios verified for WCAG AA compliance.
   ──────────────────────────────────────────── */

:root {
  /* ── Base ── */
  --background: #FAF7F0;        /* cream — 16:1 on #1F1B16 charcoal (AAA) */
  --foreground: #1F1B16;        /* deep charcoal */

  /* ── Card / elevated surface ── */
  --card: #F4F0E6;
  --card-foreground: #1F1B16;

  /* ── Muted / subdued content ── */
  --muted: #EDE8DC;
  --muted-foreground: #6B5E4E;  /* warm mid-tone, ~5:1 on cream (AA) */

  /* ── Primary brand — sage green ──
     Background / button use only.
     white-on-sage is ~3.6:1 — large-text AA only.
     Use --sage-text for body text on light backgrounds. */
  --primary: #7A8C5D;
  --primary-foreground: #FFFFFF;

  /* ── Accent — warm tan ──
     Background / decorative use ONLY.
     FAILS WCAG AA as text on cream. Use --tan-text instead. */
  --accent: #C9A876;
  --accent-foreground: #1F1B16;

  /* ── Photo-tone highlights ── */
  --gold-highlight: #D4A017;
  --gold-shadow: #8C6E10;

  /* ── Text-safe accent variants (pass WCAG AA on --background) ──
     RESEARCH.md Pitfall 3: raw sage/tan/gold FAIL as text on cream.
     These darkened variants are the safe alternatives. */
  --sage-text: #5A6B3D;         /* 5.44:1 on #FAF7F0 — AA ✓ */
  --tan-text: #8B6914;          /* 4.75:1 on #FAF7F0 — AA ✓ */

  /* ── Chrome ── */
  --border: #D9D0C0;
  --input: #EDE8DC;
  --ring: #7A8C5D;

  /* ── Radius scale ── */
  --radius: 0.5rem;
}

/* ─────────────────────────────────────────────
   Tailwind v4 theme mapping
   Maps CSS variables above to Tailwind design-token names.
   The @theme inline convention: no tailwind.config.ts needed.
   [VERIFIED: apps/platform/app/globals.css uses this exact structure]
   ──────────────────────────────────────────── */

@theme inline {
  --color-background:       var(--background);
  --color-foreground:       var(--foreground);
  --color-card:             var(--card);
  --color-card-foreground:  var(--card-foreground);
  --color-muted:            var(--muted);
  --color-muted-foreground: var(--muted-foreground);
  --color-primary:          var(--primary);
  --color-primary-foreground: var(--primary-foreground);
  --color-accent:           var(--accent);
  --color-accent-foreground: var(--accent-foreground);
  --color-border:           var(--border);
  --color-input:            var(--input);
  --color-ring:             var(--ring);
  /* Semantic extras */
  --color-sage-text:        var(--sage-text);
  --color-tan-text:         var(--tan-text);
  --color-gold-highlight:   var(--gold-highlight);
  --color-gold-shadow:      var(--gold-shadow);
  /* Radius */
  --radius-sm:  calc(var(--radius) - 2px);
  --radius-md:  var(--radius);
  --radius-lg:  calc(var(--radius) + 2px);
  --radius-xl:  calc(var(--radius) + 4px);
}

/* ─────────────────────────────────────────────
   Base styles
   Font variable injected by layout.tsx Geist import.
   ──────────────────────────────────────────── */

body {
  background: var(--background);
  color: var(--foreground);
  font-family: var(--font-geist), ui-sans-serif, system-ui, sans-serif;
  letter-spacing: 0.01em;   /* generous tracking — mirrors 2sranch.com */
  line-height: 1.6;
  -webkit-font-smoothing: antialiased;
  -moz-osx-font-smoothing: grayscale;
}

h1,
h2,
h3 {
  letter-spacing: -0.02em; /* tighter on headings — modern farm-brand convention */
  line-height: 1.2;
  font-weight: 700;
}

/* ─────────────────────────────────────────────
   Utility: visually hidden (accessible)
   ──────────────────────────────────────────── */

.sr-only {
  position: absolute;
  width: 1px;
  height: 1px;
  padding: 0;
  margin: -1px;
  overflow: hidden;
  clip: rect(0, 0, 0, 0);
  white-space: nowrap;
  border-width: 0;
}
```
