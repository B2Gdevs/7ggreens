```tsx
"use client";

import { useMemo, type ReactNode } from "react";
import { usePathname, useRouter } from "next/navigation";
import {
  VisualContextProvider,
  DevIdProvider,
  type VCSRouterAdapterValue,
} from "@portfolio/visual-context";

/**
 * SiteVisualContextProvider — 7greens VCS adapter for Next.js App Router.
 *
 * Wraps the page tree in:
 *   VisualContextProvider (router adapter — lets SiteSection resolve pathname)
 *   └─ DevIdProvider (enables dev cid inspector panel in NODE_ENV=development)
 *
 * This is a "use client" component. It is imported by app/layout.tsx (Server
 * Component) which creates the RSC boundary automatically — do NOT add
 * "use client" to layout.tsx.
 *
 * Pattern: mirrors apps/platform/components/PlatformVisualContextProvider.tsx
 * with the addition of DevIdProvider (apps/platform omits it; 7greens needs it
 * for the FOUND-04 dev-modal requirement).
 */
export function SiteVisualContextProvider({
  children,
}: {
  children: ReactNode;
}) {
  const pathname = usePathname() ?? "/";
  const nextRouter = useRouter();

  const adapter = useMemo<VCSRouterAdapterValue>(
    () => ({
      pathname,
      router: {
        push: (href) => nextRouter.push(href),
        replace: (href) => nextRouter.replace(href),
        back: () => nextRouter.back(),
        forward: () => nextRouter.forward(),
        refresh: () => nextRouter.refresh(),
        prefetch: (href) => nextRouter.prefetch(href),
      },
    }),
    [pathname, nextRouter],
  );

  return (
    <VisualContextProvider router={adapter}>
      <DevIdProvider>{children}</DevIdProvider>
    </VisualContextProvider>
  );
}
```

Key decisions (do not deviate):
1. `"use client"` at top — mandatory; VisualContextProvider and DevIdProvider use React Context.
2. `useMemo` on the adapter — prevents re-creating the object every render (same pattern as platform).
3. `DevIdProvider` wraps `children` INSIDE `VisualContextProvider` — order matters; VisualContextProvider must be the outer provider.
4. No `useEffect`, no `useState` beyond what useMemo provides.
